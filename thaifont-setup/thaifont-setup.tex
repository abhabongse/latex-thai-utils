\usepackage[varqu,varl]{inconsolata}
\usepackage{amsthm}
\usepackage[libertine]{newtxmath}
\usepackage[main=thai,thai,english]{babel}

% Complicated configuration of fonts
\makeatletter
\usepackage[resetletterform]{thaifont-setup}
\def\fontstlwg@scale{0.9}  % new feature in fonts-tlwg package
\def\letterformdefault@en{sf}  % requires 'resetletterform' option
\def\letterformdefault@th{sf}  % requires 'resetletterform' option
\def\familydefault@th@sf{laksaman}  % defaults to TH Sarabun New
\def\familydefault@th@tt{zi4}  % monospaced will always be inconsolata
\def\encodingdefault@th@tt{T1}  % make sure that inconsolata uses T1
