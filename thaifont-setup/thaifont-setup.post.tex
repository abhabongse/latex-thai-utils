\documentclass[11pt]{ltxguide}

%% Loading main utility package
\usepackage{etoolbox}

%% Setup languages, fonts, and encodings
\usepackage[LTH,T1]{fontenc}
\usepackage{libertineRoman}
\usepackage[defaultsans,scale=0.9]{opensans}
\usepackage[varqu,varl]{inconsolata}
\usepackage{amsthm}
\usepackage[libertine]{newtxmath}
\usepackage[main=thai,thai,english]{babel}
\usepackage[utf8x]{inputenc}

%% Complicated configuration of fonts
\makeatletter
\usepackage[resetletterform]{thaifont-setup}
% Scale all fonts from fonts-tlwg package (recently updated feature).
\def\fontstlwg@scale{0.9}
% Use sans-serif as default for both Latin and Thai scripts.
% This requires the option 'resetletterform' for thaifont-setup package.
\def\letterformdefault@en{sf}
\def\letterformdefault@th{sf}
% Set TH Sarabus New as default Thai sans-serif font.
\def\familydefault@th@sf{laksaman}
% Use light/semibold variant for mdseries and bfseries font in opensans.
\def\seriesdefault@en@sf@md{l}
\def\seriesdefault@en@sf@bf{sb}
% Default to inconsolata for monospaced fonts (with encoding T1)
\def\familydefault@th@tt{zi4}
\def\encodingdefault@th@tt{T1}
\makeatother

%% Set text leading
\RequirePackage{setspace}
\setstretch{1.33}

%% Document metadata
\title{แพ็ก{\wbr}เก{\wbr}จ \textlatin{\texttt{thaifont-setup}}}
\author{อาภา{\wbr}พงศ์ จันทร์{\wbr}ทอง}

\date{เวอร์ชัน 0.1\\9 ตุลาคม 2017}

\begin{document}
\maketitle

\latintext
English text
\bfseries
followed by
\thaitext
ภาษา{\wbr}ไทย{\wbr}
\mdseries
กลับ{\wbr}มา{\wbr}เป็น{\wbr}เหมือน{\wbr}เดิม{\wbr}
\latintext
and ends with English.

\end{document}
